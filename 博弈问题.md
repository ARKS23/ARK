# 博弈问题


### 题目
- **870 优势洗牌**
- ***田忌赛马问题***
- 打得过就打，**打不过就拿自己的垃圾和对方的精锐互换**
```C++
struct comp {
    bool operator() (pair<int, int> &a, pair<int, int> &b) {
        return a.first < b.first;
    }
};

/*
 * 打得过就打，打不过就拿自己的垃圾和对方的精锐互换.
 */
class Solution {
public:
    vector<int> advantageCount(vector<int>& nums1, vector<int>& nums2) {
        vector<int> result(nums1.size());
        int left = 0, right = nums1.size() - 1;
        sort(nums1.begin(), nums1.end());
        priority_queue<pair<int, int>, vector<pair<int, int>>, comp> q;
        fill(q, nums2);
        while (!q.empty()) {
            int boss = q.top().first;
            int boss_index = q.top().second;
            q.pop();
            if (nums1[right] > boss) {                          //最大的打得过boss,直接上
                result[boss_index] = nums1[right];
                right--;
            }
            else {
                result[boss_index] = nums1[left];               //打不过就用垃圾上去顶
                left++;
            }
        }
        return result;
    }

    void fill(priority_queue<pair<int, int>, vector<pair<int, int>>, comp> &q, vector<int> &nums2) {
        for (int i = 0; i < nums2.size(); ++i) 
            q.push({nums2[i], i});
    }
};
```
